    def predict(self, X_test: np.ndarray) -> np.ndarray:
    S_test = np.searchsorted(
        self.__S_valid,
        self.detector.predict(X_test),
    ) / len(self.__S_valid)

    return (
        S_test
        if type(self.detector).low_score_is_low_confidence()
        else 1.0 - S_test
    )
